1.#Query that shows sneaker name, price and size of customers by joining the sneaker brand  and category 
SELECT sneaker_name, price, size FROM sneaker_brand JOIN category WHERE customer.id=sneaker_name.id;
2. #Query to select all customer emails from the Database
SELECT emails FROM customer;
3.#Query to count the number of customer emails from the Database
SELECT COUNT(emails) FROM customer;
4.#Query that shows number of customer emails that purchased 4 pairs of sneakers
SELECT COUNT(email) FROM customers WHERE quantity = “4”;
5.#Query that  shows the customers that purchased only the brand sneaker NIKE
SELECT name FROM customers WHERE sneaker_brand =”&NIKE&”;
